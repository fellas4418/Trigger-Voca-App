        function wordCycle(data, time) {
            const star = document.getElementById('star-btn');
star.style.display = 'block';
star.disabled = false;
            document.getElementById('timer-bar').style.width = "0%";
            return new Promise(resolve => {
                const bar = document.getElementById('timer-bar');
                document.getElementById('display-word').innerText = data.word;
                document.body.style.backgroundColor = data.color;
                document.getElementById('meanings-box').innerHTML = data.mean.map(m => `<div class="meaning">${m}</div>`).join('');
                let elapsed = 0; if(timerId) clearInterval(timerId);
                timerId = setInterval(() => {
                    if(!isPaused) {
                        elapsed += 100; bar.style.width = (elapsed / time * 100) + '%';
                        const list = document.querySelectorAll('.meaning');
                        const activeIdx = Math.floor((elapsed / time) * data.mean.length);
                        list.forEach((el, idx) => { if(idx === activeIdx) el.classList.add('active'); else el.classList.remove('active'); });
                        if(elapsed >= time) { clearInterval(timerId); resolve(); }
                    }
                }, 100);
            });
        }

        function testQuestion(data, limit) {
            return new Promise(resolve => {
                document.getElementById('display-word').innerText = data.word;
document.body.style.backgroundColor = "#ffffff";
document.getElementById('star-btn').style.display = "none";
                const box = document.getElementById('test-box');
box.style.display = 'block';
                
                // 보기 무작위 셔플 [cite: 2026-03-05]
                const pool = [...data.wrong, data.mean[0]].sort(() => Math.random() - 0.5);
                box.innerHTML = "";
                pool.forEach(txt => {
                    const b = document.createElement('button');
                    b.className = 'test-option'; b.innerText = txt;
                    b.onclick = () => {
                        clearInterval(timerId);
                        const allBtns = document.querySelectorAll('.test-option');
                        allBtns.forEach(btn => {
                            btn.disabled = true;
                            if(btn.innerText === data.mean[0]) btn.classList.add('correct');
                            else if(btn === b) btn.classList.add('wrong');
                        });
                        if(txt === data.mean[0]) {

if(!inReview){
score++;
}

}
else {

saveToErrorPool(data);

if(!inReview){
retryQueue.push(data);
}

}
                        setTimeout(resolve, 1300);
                    };
                    box.appendChild(b);
                });
                let elapsed = 0; if(timerId) clearInterval(timerId);
                timerId = setInterval(() => {
                    elapsed += 100; document.getElementById('timer-bar').style.width = (100 - (elapsed / limit * 100)) + '%';
                    if(elapsed >= limit) { 
    clearInterval(timerId); 

    if(!inReview){
        saveToErrorPool(data);
    }

    document.querySelectorAll('.test-option').forEach(b => { 
        if(b.innerText === data.mean[0]) b.classList.add('correct'); 
        b.disabled = true; 
    });

    setTimeout(resolve, 1300);
}
                }, 100);
            });
        }

        function saveToErrorPool(wordObj) {
            let pool = JSON.parse(localStorage.getItem('tv_error_pool') || '[]');
            if (!pool.find(i => i.word === wordObj.word)) {
                pool.push(wordObj);
                if (pool.length > 50) pool.shift();
                localStorage.setItem('tv_error_pool', JSON.stringify(pool));
            }
        }

        function getTodayKey() {
    const d = new Date();
    const y = d.getFullYear();
    const m = String(d.getMonth() + 1).padStart(2, '0');
    const day = String(d.getDate()).padStart(2, '0');
    return `${y}-${m}-${day}`;
        }

    function getDateKey(offset) {
    const d = new Date();
    d.setDate(d.getDate() + offset);
    const y = d.getFullYear();
    const m = String(d.getMonth() + 1).padStart(2, '0');
    const day = String(d.getDate()).padStart(2, '0');
    return `${y}-${m}-${day}`;
    }

        function runCooldownCheck() {
            const stats = JSON.parse(localStorage.getItem(`tv_stats_${user.name}_${user.phone}`) || '{"sessions":0, "lastTime":0}');
            const diff = Date.now() - stats.lastTime;
            if(diff < 600000 && stats.sessions > 0) {
                const area = document.getElementById('cooldown-area'); area.style.display = 'block';
                document.getElementById('admin-reset').style.display = 'inline-block';
                const timer = setInterval(() => {
                    const rem = 600000 - (Date.now() - stats.lastTime);
                    if(rem <= 0) { clearInterval(timer); location.reload(); }
                    else area.innerText = `⏳ 휴식 중: ${Math.floor(rem / 60000)}분 ${Math.floor((rem % 60000) / 1000)}초 남음`;
                }, 1000);
            }
        }

function startCooldown(){

const area = document.getElementById('cooldown-area');

area.style.display = 'block';

let remain = SESSION_COOLDOWN;

const timer = setInterval(()=>{

    remain -= 1000;

    const m = Math.floor(remain/60000);
    const s = Math.floor((remain%60000)/1000);

    area.innerText =
    `다음 세션까지 휴식 ⏳ ${m}분 ${s}초`;

    if(remain <= 0){

        clearInterval(timer);

        area.style.display='none';

        currentSession++;

        launchMainSession();

    }

},1000);

}
